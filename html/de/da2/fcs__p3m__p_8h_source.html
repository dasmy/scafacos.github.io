89 /* This function is defined inline for maximal performance! */
90 static inline fcs_float
91 fcs_p3m_compute_near_potential(fcs_p3m_near_parameters_t params, fcs_float dist) {
92  const fcs_float alpha = params;
93  fcs_float adist = alpha * dist;
94 
95 #if FCS_P3M_USE_ERFC_APPROXIMATION
96 
97  /* approximate \f$ \exp(d^2) \mathrm{erfc}(d)\f$ by applying a formula from:
98  Abramowitz/Stegun: Handbook of Mathematical Functions, Dover
99  (9. ed.), chapter 7 */
100  fcs_float t = 1.0 / (1.0 + 0.3275911 * adist);
101  fcs_float erfc_part_ri = exp(-adist*adist) *
102  (t * (0.254829592 +
103  t * (-0.284496736 +
104  t * (1.421413741 +
105  t * (-1.453152027 +
106  t * 1.061405429)))))
107  / dist;
108 
109 #else
110 
111  /* use erf instead of erfc to fix ICC performance problems */
112  fcs_float erfc_part_ri = (1.0 - erf(adist)) / dist;
113 
114 #endif
115 
116  return erfc_part_ri;
117 
118 }

133 static inline fcs_float
134 fcs_p3m_compute_near_field(fcs_p3m_near_parameters_t params, fcs_float dist) {
135  const fcs_float alpha = params;
136  fcs_float adist = alpha * dist;
137 
138 #if FCS_P3M_USE_ERFC_APPROXIMATION
139 
140  /* approximate \f$ \exp(d^2) \mathrm{erfc}(d)\f$ by applying a formula from:
141  Abramowitz/Stegun: Handbook of Mathematical Functions, Dover
142  (9. ed.), chapter 7 */
143  fcs_float t = 1.0 / (1.0 + 0.3275911 * adist);
144  fcs_float erfc_part_ri = exp(-adist*adist) *
145  (t * (0.254829592 +
146  t * (-0.284496736 +
147  t * (1.421413741 +
148  t * (-1.453152027 +
149  t * 1.061405429)))))
150  / dist;
151  /* 1/sqrt(PI) = 0.56418958354775627928034964498 */
152  return -(erfc_part_ri + 2.0*alpha*0.56418958354775627928034964498*exp(-adist*adist))
153  / dist;
154 
155 #else
156 
157  fcs_float erfc_part_ri = (1.0 - erf(adist)) / dist; /* use erf instead of erfc to fix ICC performance problems */
158  /* 1/sqrt(PI) = 0.56418958354775627928034964498 */
159  return -(erfc_part_ri + 2.0*alpha*0.56418958354775627928034964498*exp(-adist*adist))
160  / dist;
161 
162 #endif
